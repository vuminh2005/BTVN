#include <bits/stdc++.h>

using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    int arr[n];
    
    int count = 0;
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
        for (int j = 0; j < i; j++) {
            int sum = 0;
            sum = arr[i] + arr[j];
            if (sum % k == 0) count++;
        }
    }
    cout << count;
}
